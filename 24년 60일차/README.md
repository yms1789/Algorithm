      if (cur === start) {
        if (arr.length === 0) {
          result.push(cur);
        } else {
          result.push(start, ...arr);
        }
        return;
      }
      if (!visited[cur]) {
        arr.push(cur);
        visited[cur] = true;
        dfs(start, team[cur][0], team, arr, visited, result);
